<!DOCTYPE html>
<html lang="ru">
<head>
    <meta charset="UTF-8">
    <title>Змейка</title>
    <style>
        body {
            margin: 0;
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
            background: linear-gradient(135deg, #0f0f1a, #1a0f0f);
            font-family: 'Segoe UI', Tahoma, Geneva, Verdana, sans-serif;
            color: white;
            overflow: hidden;
        }
        canvas {
            border: 4px solid #fff;
            background: radial-gradient(circle at center, #000, #1a1a2f);
            box-shadow: 0 0 30px rgba(255,255,255,0.1);
        }
        #score {
            position: absolute;
            top: 20px;
            left: 20px;
            font-size: 24px;
            font-weight: bold;
            text-shadow: 0 0 5px #fff;
            z-index: 10;
        }
        .game-over {
            position: absolute;
            top: 50%;
            left: 50%;
            transform: translate(-50%, -50%);
            font-size: 48px;
            text-align: center;
            opacity: 0;
            transition: opacity 1s ease-out;
        }
        .game-over.show {
            opacity: 1;
            animation: pulse 2s infinite alternate;
        }
        @keyframes pulse {
            0% { transform: translate(-50%, -50%) scale(1); }
            100% { transform: translate(-50%, -50%) scale(1.1); }
        }
    </style>
</head>
<body>
    <div id="score">Score: 0</div>
    <canvas id="game" width="400" height="400"></canvas>
    <div class="game-over" id="gameOver">Game Over! Score: <span id="finalScore">0</span></div>
    <script>
        const canvas = document.getElementById('game');
        const ctx = canvas.getContext('2d');
        const gridSize = 20;
        const tileSize = 20;

        let snake = [[10, 10], [9, 10], [8, 10]];
        let direction = { x: 0, y: 1 };
        let food = null;
        let score = 0;
        let gameRunning = false;
        let gameOverElement = document.getElementById('gameOver');
        let finalScoreElement = document.getElementById('finalScore');

        function draw() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // Draw background gradient
            ctx.fillStyle = 'radial-gradient(circle at center, #000, #1a1a2f)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Draw snake with shadow
            for (let segment of snake) {
                ctx.shadowColor = '#4caf50';
                ctx.shadowBlur = 10;
                ctx.fillStyle = 'rgba(76, 175, 80, 0.9)';
                ctx.fillRect(segment[0] * tileSize, segment[1] * tileSize, tileSize, tileSize);
            }

            // Draw food with glow effect
            ctx.shadowColor = '#ff4081';
            ctx.shadowBlur = 15;
            ctx.fillStyle = 'rgba(255, 64, 130, 0.8)';
            ctx.fillRect(food[0] * tileSize, food[1] * tileSize, tileSize, tileSize);
        }

        function generateFood() {
            let newFood;
            while (true) {
                let x = Math.floor(Math.random() * 20);
                let y = Math.floor(Math.random() * 20);
                if (!snake.some(segment => segment[0] === x && segment[1] === y)) {
                    newFood = [x, y];
                    break;
                }
            }
            food = newFood;
        }

        function moveSnake() {
            const head = [snake[0][0] + direction.x, snake[0][1] + direction.y];
            
            // Check collision with walls
            if (head[0] < 0 || head[0] >= 20 || head[1] < 0 || head[1] >= 20) {
                endGame();
                return;
            }
            
            // Check collision with self
            const body = snake.slice(1);
            if (body.some(segment => segment[0] === head[0] && segment[1] === head[1])) {
                endGame();
                return;
            }

            // Move snake
            snake.unshift(head);

            // Check for food
            if (head[0] === food[0] && head[1] === food[1]) {
                score++;
                generateFood();
                // Food glow effect
                ctx.fillStyle = 'rgba(255, 64, 130, 0.8)';
                ctx.fillRect(food[0] * tileSize, food[1] * tileSize, tileSize, tileSize);
            } else {
                snake.pop();
            }

            draw();
        }

        function endGame() {
            gameRunning = false;
            gameOverElement.classList.add('show');
            finalScoreElement.textContent = score;
            // Reset canvas
            ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
        }

        document.addEventListener('keydown', function(e) {
            const key = e.key;
            if (key === 'ArrowUp' || key === 'w') {
                if (direction.y !== 1) direction = { x: 0, y: -1 };
            } else if (key === 'ArrowDown' || key === 's') {
                if (direction.y !== -1) direction = { x: 0, y: 1 };
            } else if (key === 'ArrowLeft' || key === 'a') {
                if (direction.x !== 1) direction = { x: -1, y: 0 };
            } else if (key === 'ArrowRight' || key === 'd') {
                if (direction.x !== -1) direction = { x: 1, y: 0 };
            } else if (key === 'Enter') {
                if (!gameRunning) {
                    snake = [[10, 10], [9, 10], [8, 10]];
                    direction = { x: 0, y: 1 };
                    score = 0;
                    gameRunning = true;
                    generateFood();
                    gameOverElement.classList.remove('show');
                    draw();
                    gameLoop();
                }
            }
        });

        function gameLoop() {
            if (!gameRunning) return;
            moveSnake();
            setTimeout(gameLoop, 100);
        }

        // Initialize
        generateFood();
        gameRunning = true;
        document.getElementById('score').innerText = 'Score: ' + score;
        gameLoop();
    </script>
</body>
</html>
