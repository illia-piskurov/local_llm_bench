```html
<!DOCTYPE html>
<html lang="ru">
<head>
<meta charset="UTF-8">
<title>Змейка</title>
<style>
  body{margin:0;background:#111;color:#fff;font-family:sans-serif;text-align:center;}
  #game{border-radius:10px;box-shadow:0 0 20px rgba(255,255,255,.2);background:#222;margin:auto;display:block;}
  #info{margin-top:10px;font-size:18px;}
  .overlay{
    position:absolute;top:0;left:0;width:100%;height:100%;
    background:rgba(0,0,0,.8);
    display:flex;align-items:center;justify-content:center;
    flex-direction:column;color:#fff;font-size:36px;
    opacity:0;pointer-events:none;
    transition:opacity .5s ease;
  }
  .overlay.show{opacity:1;}
  canvas{display:block;background:#111;border-radius:10px;}
</style>
</head>
<body>
<canvas id="game" width="400" height="400"></canvas>
<div id="info">Счёт: <span id="score">0</span></div>

<div class="overlay" id="over">
  Game Over<br><span style="font-size:24px;">Нажмите любую клавишу для перезапуска.</span>
</div>

<script>
(() => {
const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');
const scoreEl = document.getElementById('score');
const overlay = document.getElementById('over');

const cellSize = 20;
const rows = canvas.height / cellSize;
const cols = canvas.width / cellSize;

let snake = [{x:5,y:5}];
let dir = {x:1,y:0};
let food = {};
let gameOver = false;
let score = 0;
let lastTime = 0;
let speed = 150; // ms per move
let particles = [];

function placeFood(){
  let ok=false;
  while(!ok){
    const fx=Math.floor(Math.random()*cols);
    const fy=Math.floor(Math.random()*rows);
    if (!snake.some(s=>s.x===fx&&s.y===fy)) {food={x:fx,y:fy};ok=true;}
  }
}
function drawCell(x,y,color,alpha=1){
  ctx.globalAlpha=alpha;
  ctx.fillStyle=color;
  ctx.shadowBlur=10;ctx.shadowColor='rgba(0,255,0,.6)';
  ctx.fillRect(x*cellSize+2,y*cellSize+2,(cellSize-4)*(dir.x===0?1:1),(cellSize-4));
  ctx.globalAlpha=1;
}
function update(){
  const head = {...snake[0]};
  head.x += dir.x;
  head.y += dir.y;

  if(head.x<0||head.x>=cols||head.y<0||head.y>=rows){end();return;}
  if(snake.some(s=>s.x===head.x&&s.y===head.y)){end();return;}

  snake.unshift(head);
  if(head.x===food.x && head.y===food.y){
    score++;scoreEl.textContent=score;
    createParticles(head.x,head.y);
    placeFood();
  }else{
    snake.pop();
  }
}
function draw(){
  ctx.clearRect(0,0,canvas.width,canvas.height);
  // food fade
  if(food){drawCell(food.x,food.y,'red',.8);}
  snake.forEach((s,i)=>drawCell(s.x,s.y,i===0?'lime':'green'));
  particles.forEach(p=>{p.life--;if(p.life>0)drawParticle(p);});
  particles = particles.filter(p=>p.life>0);
}
function drawParticle(p){
  ctx.globalAlpha=p.life/20;
  ctx.fillStyle='yellow';
  ctx.beginPath();
  ctx.arc((p.x+0.5)*cellSize,(p.y+0.5)*cellSize,p.size,0,Math.PI*2);
  ctx.fill();
  ctx.globalAlpha=1;
}
function createParticles(x,y){
  for(let i=0;i<15;i++){
    particles.push({
      x:x+Math.random()*0.8,
      y:y+Math.random()*0.8,
      size:Math.random()*3+2,
      life:20
    });
  }
}
function loop(time){
  if(!lastTime) lastTime=time;
  const delta = time-lastTime;
  if(delta>speed){update();lastTime=time;}
  draw();
  requestAnimationFrame(loop);
}
function start(){
  snake=[{x:5,y:5}];
  dir={x:1,y:0};
  score=0;scoreEl.textContent=0;
  gameOver=false;
  overlay.classList.remove('show');
  placeFood();
  lastTime=0;
  requestAnimationFrame(loop);
}
function end(){
  gameOver=true;
  overlay.classList.add('show');
}
document.addEventListener('keydown',e=>{
  if(gameOver){start();return;}
  const key=e.key;
  if(key==='ArrowUp'||key==='w'){if(dir.y!==1)dir={x:0,y:-1};}
  else if(key==='ArrowDown'||key==='s'){if(dir.y!==-1)dir={x:0,y:1};}
  else if(key==='ArrowLeft'||key==='a'){if(dir.x!==1)dir={x:-1,y:0};}
  else if(key==='ArrowRight'||key==='d'){if(dir.x!==-1)dir={x:1,y:0};}
});
start();
})();
</script>
</body>
</html>
```
