<body>

    <h1>Змейка</h1>

    <div class="ui-panel">
        <div class="score-box">Счёт: <span id="scoreEl">0</span></div>
        <div class="high-score-box">Рекорд: <span id="highScoreEl">0</span></div>
    </div>

    <div class="game-wrapper">
        <canvas id="gameCanvas" width="400" height="400"></canvas>
        
        <div id="gameOverOverlay">
            <div id="gameOverText">Game Over</div>
            <button class="btn-restart" onclick="resetGame()">Играть снова</button>
        </div>
    </div>

    <div class="controls-hint">
        Управление: <span class="key">WASD</span> или <span class="key">Стрелки</span> | 
        Пауза: <span class="key">Пробел</span>
    </div>

    <script>
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        const scoreEl = document.getElementById('scoreEl');
        const highScoreEl = document.getElementById('highScoreEl');
        const gameOverOverlay = document.getElementById('gameOverOverlay');
        const gameOverText = document.getElementById('gameOverText');

        // Настройки игры
        const gridSize = 20; // Размер одной клетки
        const tileCount = canvas.width / gridSize; // Количество клеток по ширине (20)
        
        let score = 0;
        let highScore = localStorage.getItem('snakeHighScore') || 0;
        highScoreEl.innerText = highScore;

        let velocityX = 0;
        let velocityY = 0;
        
        let snake = [];
        let food = { x: 15, y: 15 };
        
        let gameInterval;
        let isGameRunning = false;
        let isPaused = false;

        // Инициализация игры
        function initGame() {
            snake = [
                { x: 10, y: 10 },
                { x: 9, y: 10 },
                { x: 8, y: 10 }
            ];
            score = 0;
            scoreEl.innerText = score;
            velocityX = 1; // Движение вправо по умолчанию
            velocityY = 0;
            placeFood();
            isGameRunning = true;
            isPaused = false;
            gameOverOverlay.classList.remove('active');
            
            if (gameInterval) clearInterval(gameInterval);
            gameInterval = setInterval(gameLoop, 100); // Скорость игры (100мс)
        }

        // Основной цикл игры
        function gameLoop() {
            if (isPaused) return;

            update();
            draw();
        }

        // Логика обновления
        function update() {
            // Вычисляем новую позицию головы
            const head = { x: snake[0].x + velocityX, y: snake[0].y + velocityY };

            // Проверка столкновения со стенами
            if (head.x < 0 || head.x >= tileCount || head.y < 0 || head.y >= tileCount) {
                gameOver();
                return;
            }

            // Проверка столкновения с собой
            for (let i = 0; i < snake.length; i++) {
                if (head.x === snake[i].x && head.y === snake[i].y) {
                    gameOver();
                    return;
                }
            }

            // Добавляем новую голову
            snake.unshift(head);

            // Проверка поедания еды
            if (head.x === food.x && head.y === food.y) {
                score += 10;
                scoreEl.innerText = score;
                placeFood();
                // Не удаляем хвост, змейка растет
            } else {
                // Удаляем хвост, змейка движется
                snake.pop();
            }
        }

        // Отрисовка
        function draw() {
            // Очистка канваса
            ctx.fillStyle = '#000';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Отрисовка еды
            ctx.fillStyle = '#f44336';
            ctx.beginPath();
            ctx.arc(
                food.x * gridSize + gridSize / 2, 
                food.y * gridSize + gridSize / 2, 
                gridSize / 2 - 2, 0, Math.PI * 2
            );
            ctx.fill();

            // Отрисовка змейки
            snake.forEach((part, index) => {
                // Голова другого цвета
                if (index === 0) {
                    ctx.fillStyle = '#81c784';
                } else {
                    ctx.fillStyle = '#4caf50';
                }
                
                ctx.fillRect(
                    part.x * gridSize + 1, 
                    part.y * gridSize + 1, 
                    gridSize - 2, 
                    gridSize - 2
                );
            });
        }

        // Размещение еды в случайной точке
        function placeFood() {
            let validPosition = false;
            while (!validPosition) {
                food.x = Math.floor(Math.random() * tileCount);
                food.y = Math.floor(Math.random() * tileCount);

                // Проверяем, чтобы еда не появилась внутри змейки
                validPosition = true;
                for (let part of snake) {
                    if (part.x === food.x && part.y === food.y) {
                        validPosition = false;
                        break;
                    }
                }
            }
        }

        // Конец игры
        function gameOver() {
            isGameRunning = false;
            clearInterval(gameInterval);
            
            if (score > highScore) {
                highScore = score;
                localStorage.setItem('snakeHighScore', highScore);
                highScoreEl.innerText = highScore;
                gameOverText.innerText = "Новый рекорд!";
            } else {
                gameOverText.innerText = "Game Over";
            }
            
            gameOverOverlay.classList.add('active');
        }

        // Сброс игры (кнопка)
        function resetGame() {
            initGame();
        }

        // Обработка клавиатуры
        document.addEventListener('keydown', (e) => {
            // Управление направлением
            switch (e.key) {
                case 'ArrowUp':
                case 'w':
                case 'W':
                    if (velocityY !== 1) { velocityX = 0; velocityY = -1; }
                    break;
                case 'ArrowDown':
                case 's':
                case 'S':
                    if (velocityY !== -1) { velocityX = 0; velocityY = 1; }
                    break;
                case 'ArrowLeft':
                case 'a':
                case 'A':
                    if (velocityX !== 1) { velocityX = -1; velocityY = 0; }
                    break;
                case 'ArrowRight':
                case 'd':
                case 'D':
                    if (velocityX !== -1) { velocityX = 1; velocityY = 0; }
                    break;
                case ' ':
                    // Пауза
                    if (isGameRunning) {
                        isPaused = !isPaused;
                    }
                    break;
            }
        });

        // Запуск при загрузке
        initGame();

    </script>
</body>
</html>
