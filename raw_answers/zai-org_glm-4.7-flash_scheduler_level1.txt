from collections import deque

def topo_sort(tasks: dict[str, list[str]]) -> list[str] | None:
    # Собираем множество всех уникальных задач
    all_tasks = set(tasks.keys())
    for deps in tasks.values():
        all_tasks.update(deps)

    # Вычисляем количество зависимостей (in-degree) для каждой задачи
    indegree = {task: 0 for task in all_tasks}
    for task, deps in tasks.items():
        for dep in deps:
            indegree[dep] += 1

    # Инициализируем очередь задач, которые можно выполнить сразу (in-degree == 0)
    queue = deque([task for task in all_tasks if indegree[task] == 0])

    result = []
    while queue:
        current = queue.popleft()
        result.append(current)

        # Находим задачи, которые зависят от текущей
        for task, deps in tasks.items():
            if current in deps:
                indegree[task] -= 1
                if indegree[task] == 0:
                    queue.append(task)

    # Если количество обработанных задач не равно общему количеству,
    # значит, есть цикл, и топологическая сортировка невозможна
    if len(result) != len(all_tasks):
        return None

    return result
