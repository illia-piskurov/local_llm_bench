    <style>
        body {

        #game-container {

        }

        canvas {
            background-color: #0f0f0f;
            display: block;
            /* Легкая сетка на фоне */
            background-image: 
                linear-gradient(#222 1px, transparent 1px),
                linear-gradient(90deg, #222 1px, transparent 1px);
            background-size: 20px 20px; /* Размер клетки сетки совпадает с логическим размером */
        }

        #score-board {
            font-size: 24px;
            margin-bottom: 15px;
            text-shadow: 0 0 10px rgba(255, 255, 255, 0.5);
            font-weight: bold;
        }

        #game-over-screen {
            position: absolute;
            top: 0;
            left: 0;
            width: 100%;
            height: 100%;
            background-color: rgba(0, 0, 0, 0.85);
            backdrop-filter: blur(4px);

        }

        @keyframes pulseText {
            0% { transform: scale(1); }
            50% { transform: scale(1.05); }
            100% { transform: scale(1); }

<body>

    <div id="score-board">Счёт: <span id="score">0</span></div>

    <div id="game-container">
        <canvas id="gameCanvas" width="400" height="400"></canvas>
        
        <div id="game-over-screen">
            <h1>Game Over</h1>
            <p>Ваш счёт: <span id="final-score">0</span></p>
            <p style="margin-top: 20px; color: #f1c40f; text-shadow: 0 0 5px orange;">Нажмите любую клавишу для рестарта</p>
        </div>

        <div class="controls-hint">Управление: Стрелки или WASD</div>
    </div>

    <script>
        // --- Конфигурация игры ---
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        const scoreElement = document.getElementById('score');
        const finalScoreElement = document.getElementById('final-score');
        const gameOverScreen = document.getElementById('game-over-screen');

        const TILE_SIZE = 20; 
        const TILE_COUNT = canvas.width / TILE_SIZE; 
        
        // --- Состояние игры ---
        let score = 0;
        let snake = [];
        let food = { x: 0, y: 0 };
        let velocity = { x: 0, y: 0 };
        let nextVelocity = { x: 0, y: 0 }; 
        let gameInterval;
        let isGameRunning = false;

        // --- Система частиц (визуальные эффекты) ---
        let particles = [];

        function initGame() {
            snake = [
                { x: 10, y: 10 }, 
                { x: 9, y: 10 },
                { x: 8, y: 10 }
            ];
            score = 0;
            velocity = { x: 1, y: 0 }; 
            nextVelocity = { x: 1, y: 0 };
            particles = []; // Очистка частиц
            scoreElement.innerText = score;
            gameOverScreen.classList.remove('active');
            
            spawnFood();
            
            if (gameInterval) clearInterval(gameInterval);
            gameInterval = setInterval(gameLoop, 100); 
            isGameRunning = true;
        }

        // --- Основной цикл ---
        function gameLoop() {
            update();
            draw();
        }

        // --- Логика обновления ---
        function update() {
            velocity = nextVelocity;

            const head = { x: snake[0].x + velocity.x, y: snake[0].y + velocity.y };

            // Проверка столкновения со стенами
            if (head.x < 0 || head.x >= TILE_COUNT || head.y < 0 || head.y >= TILE_COUNT) {
                gameOver();
                return;
            }

            // Проверка столкновения с хвостом
            for (let i = 0; i < snake.length; i++) {
                if (head.x === snake[i].x && head.y === snake[i].y) {
                    gameOver();
                    return;
                }
            }

            snake.unshift(head);

            // Проверка: съели ли еду?
            if (head.x === food.x && head.y === food.y) {
                score += 10;
                scoreElement.innerText = score;
                
                // Создаем эффект частиц
                createParticles(food.x, food.y);
                
                spawnFood();
            } else {
                snake.pop();
            }
        }

        // --- Отрисовка ---
        function draw() {
            // Очистка экрана (черный цвет)
            ctx.fillStyle = '#0f0f0f';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Рисуем еду (круг с пульсацией)
            const time = Date.now() / 200;
            const pulse = Math.sin(time) * 1.5 + TILE_SIZE/2;
            
            ctx.fillStyle = '#ff4757';
            ctx.shadowBlur = 15;
            ctx.shadowColor = '#ff4757';
            ctx.beginPath();
            ctx.arc(
                food.x * TILE_SIZE + TILE_SIZE / 2, 
                food.y * TILE_SIZE + TILE_SIZE / 2, 
                pulse - 2, 0, Math.PI * 2
            );
            ctx.fill();
            ctx.shadowBlur = 0; // Сброс тени

            // Рисуем змейку (сегменты)
            for (let i = 0; i < snake.length; i++) {
                const part = snake[i];
                
                // Голова отличается цветом
                if (i === 0) {
                    ctx.fillStyle = '#2ed573'; 
                    ctx.shadowBlur = 10;
                    ctx.shadowColor = '#2ed573';
                } else {
                    // Градиент для тела хвоста
                    const alpha = 1 - (i / snake.length) * 0.6;
                    ctx.fillStyle = `rgba(46, 213, 115, ${alpha})`; 
                    ctx.shadowBlur = 0;
                }

                // Рисуем сегмент как скругленный прямоугольник
                const x = part.x * TILE_SIZE + 1;
                const y = part.y * TILE_SIZE + 1;
                const size = TILE_SIZE - 2;
                
                ctx.beginPath();
                ctx.roundRect(x, y, size, size, 4); // Скругление углов
                ctx.fill();
            }

            // Рисуем частицы
            drawParticles();
        }

        // --- Система частиц ---
        function createParticles(x, y) {
            const centerX = x * TILE_SIZE + TILE_SIZE / 2;
            const centerY = y * TILE_SIZE + TILE_SIZE / 2;
            
            for (let i = 0; i < 15; i++) {
                particles.push({
                    x: centerX,
                    y: centerY,
                    vx: (Math.random() - 0.5) * 4, // Случайная скорость X
                    vy: (Math.random() - 0.5) * 4, // Случайная скорость Y
                    life: 1.0, // Жизнь частицы от 1 до 0
                    color: `hsl(${Math.random() * 60 + 30}, 100%, 50%)` // Оранжево-желтые цвета
                });
            }
        }

        function drawParticles() {
            for (let i = particles.length - 1; i >= 0; i--) {
                let p = particles[i];
                
                ctx.fillStyle = p.color;
                ctx.globalAlpha = p.life; // Прозрачность зависит от жизни
                
                ctx.beginPath();
                ctx.arc(p.x, p.y, 3, 0, Math.PI * 2); // Маленькие точки
                ctx.fill();

                ctx.globalAlpha = 1.0; // Сброс прозрачности

                // Обновление состояния частицы
                p.x += p.vx;
                p.y += p.vy;
                p.life -= 0.05; // Частица исчезает

                if (p.life <= 0) {
                    particles.splice(i, 1);
                }
            }
        }

        // --- Спавн еды ---
        function spawnFood() {
            food.x = Math.floor(Math.random() * TILE_COUNT);
            food.y = Math.floor(Math.random() * TILE_COUNT);

            for (let part of snake) {
                if (part.x === food.x && part.y === food.y) {
                    spawnFood(); 
                }
            }
        }

        // --- Game Over ---
        function gameOver() {
            isGameRunning = false;
            clearInterval(gameInterval);
            finalScoreElement.innerText = score;
            gameOverScreen.classList.add('active');
        }

        // --- Обработка ввода ---
        document.addEventListener('keydown', (e) => {
            if (!isGameRunning && gameOverScreen.classList.contains('active')) {
                initGame();
                return;
            }

            switch (e.key) {
                case 'ArrowUp':
                case 'w':
                case 'W':
                    if (velocity.y !== 1) nextVelocity = { x: 0, y: -1 }; 
                    break;
                case 'ArrowDown':
                case 's':
                case 'S':
                    if (velocity.y !== -1) nextVelocity = { x: 0, y: 1 };
                    break;
                case 'ArrowLeft':
                case 'a':
                case 'A':
                    if (velocity.x !== 1) nextVelocity = { x: -1, y: 0 };
                    break;
                case 'ArrowRight':
                case 'd':
                case 'D':
                    if (velocity.x !== -1) nextVelocity = { x: 1, y: 0 };
                    break;
            }
        });

        initGame();

    </script>
</body>
</html>
