    <h1>Neon Snake</h1>

    <div class="ui-panel">
        <div class="score-box">Счёт: <span id="scoreEl">0</span></div>

    <script>
        const canvas = document.getElementById('gameCanvas');
        const ctx = canvas.getContext('2d');
        const scoreEl = document.getElementById('scoreEl');
        const highScoreEl = document.getElementById('highScoreEl');
        const gameOverOverlay = document.getElementById('gameOverOverlay');
        const gameOverText = document.getElementById('gameOverText');
        const gameContainer = document.querySelector('.game-container');

        // Настройки игры
        const gridSize = 20;
        const tileCount = canvas.width / gridSize;
        
        let score = 0;
        let highScore = localStorage.getItem('snakeHighScore') || 0;
        highScoreEl.innerText = highScore;

        let velocityX = 0;
        let velocityY = 0;
        
        let snake = [];
        let food = { x: 15, y: 15 };
        
        let lastTime = 0;
        let gameSpeed = 100; // мс между кадрами змейки
        let gameRunning = false;
        let isPaused = false;
        
        // Система частиц
        let particles = [];

        class Particle {
            constructor(x, y, color) {
                this.x = x;
                this.y = y;
                this.color = color;
                // Случайная скорость разлета
                this.vx = (Math.random() - 0.5) * 8;
                this.vy = (Math.random() - 0.5) * 8;
                this.life = 1.0; // Жизнь частицы (от 1 до 0)
                this.decay = Math.random() * 0.03 + 0.02; // Скорость исчезновения
                this.size = Math.random() * 4 + 2;
            }

            update() {
                this.x += this.vx;
                this.y += this.vy;
                this.life -= this.decay;
                this.size *= 0.95; // Частицы уменьшаются
            }

            draw(ctx) {
                ctx.save();
                ctx.globalAlpha = this.life;
                ctx.fillStyle = this.color;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
                ctx.restore();
            }
        }

        function initGame() {
            snake = [
                { x: 10, y: 10 },
                { x: 9, y: 10 },
                { x: 8, y: 10 }
            ];
            score = 0;
            scoreEl.innerText = score;
            velocityX = 1; 
            velocityY = 0;
            placeFood();
            particles = [];
            gameRunning = true;
            isPaused = false;
            gameOverOverlay.classList.remove('active');
            gameContainer.classList.remove('flash');
            
            lastTime = performance.now();
            requestAnimationFrame(gameLoop);
        }

        function gameLoop(currentTime) {
            if (!gameRunning) return;

            requestAnimationFrame(gameLoop);

            const secondsSinceLastRender = (currentTime - lastTime);

            if (secondsSinceLastRender < gameSpeed) return;

            lastTime = currentTime;

            if (!isPaused) {
                update();
                draw();
            }
        }

        function update() {
            // Вычисляем новую позицию головы
            const head = { x: snake[0].x + velocityX, y: snake[0].y + velocityY };

            // Проверка столкновения со стенами
            if (head.x < 0 || head.x >= tileCount || head.y < 0 || head.y >= tileCount) {
                gameOver();
                return;
            }

            // Проверка столкновения с собой
            for (let i = 0; i < snake.length; i++) {
                if (head.x === snake[i].x && head.y === snake[i].y) {
                    gameOver();
                    return;
                }
            }

            // Добавляем новую голову
            snake.unshift(head);

            // Проверка поедания еды
            if (head.x === food.x && head.y === food.y) {
                score += 10;
                scoreEl.innerText = score;
                createExplosion(food.x * gridSize + gridSize/2, food.y * gridSize + gridSize/2);
                placeFood();
                // Визуальный эффект вспышки
                gameContainer.classList.remove('flash');
                void gameContainer.offsetWidth; // trigger reflow
                gameContainer.classList.add('flash');
            } else {
                snake.pop();
            }

            // Обновляем частицы
            particles.forEach((p, index) => {
                p.update();
                if (p.life <= 0) particles.splice(index, 1);
            });
        }

        function draw() {
            // Очистка канваса
            ctx.clearRect(0, 0, canvas.width, canvas.height);

            // Отрисовка частиц (под змейкой)
            particles.forEach(p => p.draw(ctx));

            // Отрисовка еды
            ctx.shadowBlur = 15;
            ctx.shadowColor = 'var(--food-color)';
            ctx.fillStyle = 'var(--food-color)';
            ctx.beginPath();
            // Еда немного пульсирует визуально (в логике не нужно, просто для красоты)
            const pulse = Math.sin(Date.now() / 200) * 2;
            ctx.arc(
                food.x * gridSize + gridSize / 2, 
                food.y * gridSize + gridSize / 2, 
                (gridSize / 2 - 2) + pulse, 0, Math.PI * 2
            );
            ctx.fill();
            ctx.shadowBlur = 0; // Сброс тени

            // Отрисовка змейки
            snake.forEach((part, index) => {
                // Голова светится ярче
                if (index === 0) {
                    ctx.shadowBlur = 20;
                    ctx.shadowColor = 'var(--snake-glow)';
                    ctx.fillStyle = '#fff'; // Белый центр головы
                } else {
                    ctx.shadowBlur = 10;
                    ctx.shadowColor = 'var(--snake-glow)';
                    // Градиент для тела змейки
                    ctx.fillStyle = 'var(--snake-color)';
                }
                
                // Рисуем сегмент с небольшим отступом для эффекта "клеток"
                ctx.fillRect(
                    part.x * gridSize + 1, 
                    part.y * gridSize + 1, 
                    gridSize - 2, 
                    gridSize - 2
                );
            });
            ctx.shadowBlur = 0;
        }

        function createExplosion(x, y) {
            const colors = ['#ff0055', '#ff9900', '#ffff00', '#ffffff'];
            for (let i = 0; i < 12; i++) {
                const color = colors[Math.floor(Math.random() * colors.length)];
                particles.push(new Particle(x, y, color));
            }
        }

        function placeFood() {
            let validPosition = false;
            while (!validPosition) {
                food.x = Math.floor(Math.random() * tileCount);
                food.y = Math.floor(Math.random() * tileCount);

                validPosition = true;
                for (let part of snake) {
                    if (part.x === food.x && part.y === food.y) {
                        validPosition = false;
                        break;
                    }
                }
            }
        }

        function gameOver() {
            gameRunning = false;
            
            if (score > highScore) {
                highScore = score;
                localStorage.setItem('snakeHighScore', highScore);
                highScoreEl.innerText = highScore;
                gameOverText.innerText = "Новый рекорд!";
                gameOverText.style.color = "#00ff88";
            } else {
                gameOverText.innerText = "Game Over";
                gameOverText.style.color = "#ff0055";
            }
            
            gameOverOverlay.classList.add('active');
        }

        function resetGame() {
            initGame();
        }

        document.addEventListener('keydown', (e) => {
            // Управление направлением
            switch(e.key) {
                case 'ArrowUp':
                case 'w':
                case 'W':
                    if (velocityY !== 1) { velocityX = 0; velocityY = -1; }
                    break;
                case 'ArrowDown':
                case 's':
                case 'S':
                    if (velocityY !== -1) { velocityX = 0; velocityY = 1; }
                    break;
                case 'ArrowLeft':
                case 'a':
                case 'A':
                    if (velocityX !== 1) { velocityX = -1; velocityY = 0; }
                    break;
                case 'ArrowRight':
                case 'd':
                case 'D':
                    if (velocityX !== -1) { velocityX = 1; velocityY = 0; }
                    break;
                case ' ':
                    if (gameRunning) {
                        isPaused = !isPaused;
                    }
                    break;
            }
        });

        initGame();

    </script>
</body>
