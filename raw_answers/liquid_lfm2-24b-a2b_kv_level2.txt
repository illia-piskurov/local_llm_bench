class InMemoryKV:
    def __init__(self):
        self.global_store = {}
        self.transactions = []  # stack of dicts for open transactions
        self.watchers = {}      # key -> (old_value, watcher_thread)

    def _find_transaction(self, depth):
        if depth == 0:
            return self.transactions[-1] if self.transactions else None
        for i in range(len(self.transactions) - depth, -1, -1):
            if len(self.transactions[i]) > i:
                return self.transactions[i][depth]
        return None

    def _get_current_value(self, key):
        # Search from innermost to global
        for txn in reversed(self.transactions):
            if key in txn:
                return txn[key]
        return self.global_store.get(key)

    def set(self, key, value):
        txn = self._find_transaction(len(self.transactions))
        if txn is None:
            txn = {}
            self.transactions.append(txn)
        old_val = txn.get(key)
        txn[key] = value
        # Notify watchers
        if key in self.watchers:
            old, _ = self.watchers[key]
            new = value
            print(f"WATCH {key} {old} -> {new}")
        self.watchers.pop(key, None)  # remove after notify

    def get(self, key):
        for txn in reversed(self.transactions):
            if key in txn:
                return txn[key]
        return "NULL"

    def delete(self, key):
        txn = self._find_transaction(len(self.transactions))
        if txn is not None and key in txn:
            old_val = txn[key]
            del txn[key]
            # Notify watchers
            if key in self.watchers:
                self.watchers[key] = (old_val, None)
            else:
                self.watchers[key] = (None, None)

    def begin(self):
        self.transactions.append({})

    def commit(self):
        if not self.transactions:
            return "NO TRANSACTION"
        current = self.transactions.pop()
        if not self.transactions:
            for k, v in current.items():
                self.global_store[k] = v
        else:
            parent = self.transactions[-1]
            for k, v in current.items():
                parent[k] = v
        # Clear watchers after commit (they are now part of global)
        self.watchers = {}
        return None

    def rollback(self):
        if not self.transactions:
            return "NO TRANSACTION"
        txn = self.transactions.pop()
        # Discard changes
        return None


def run(program: str):
    kv = InMemoryKV()
    output = []
    lines = program.strip().split('\n')
    for line in lines:
        if not line.strip():
            continue
        parts = line.split()
        cmd = parts[0]
        if cmd == "SET":
            key, value = parts[1], parts[2]
            kv.set(key, value)
        elif cmd == "GET":
            key = parts[1]
            output.append(str(kv.get(key)))
        elif cmd == "DELETE":
            key = parts[1]
            kv.delete(key)
        elif cmd == "BEGIN":
            kv.begin()
        elif cmd == "COUNT":
            value = int(parts[1])
            cnt = 0
            for k in kv.global_store:
                if kv.get(k) == value:
                    cnt += 1
            output.append(str(cnt))
        elif cmd == "WATCH":
            key = parts[1]
            old_val = kv.get(key)
            # Notify current value (may be None)
            new_val = kv.get(key)
            if old_val is None:
                old_str = "NULL"
            else:
                old_str = str(old_val)
            if new_val is None:
                new_str = "NULL"
            else:
                new_str = str(new_val)
            output.append(f"WATCH {key} {old_str} -> {new_str}")
        elif cmd == "COMMIT":
            res = kv.commit()
            if res is not None:
                output.append(res)
        elif cmd == "ROLLBACK":
            res = kv.rollback()
            if res is not None:
                output.append(res)
    return output
