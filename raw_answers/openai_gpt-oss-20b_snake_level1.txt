```html
<!DOCTYPE html>
<html lang="ru">
<head>
<meta charset="UTF-8">
<title>Змейка</title>
<style>
  body{margin:0;background:#222;color:#fff;font-family:sans-serif;text-align:center;}
  #game{border:2px solid #555;margin:auto;display:block;background:#111;}
  #info{margin-top:10px;}
  button{padding:5px 15px;font-size:16px;}
</style>
</head>
<body>
<canvas id="game" width="400" height="400"></canvas>
<div id="info">Счёт: <span id="score">0</span></div>
<script>
(() => {
const canvas = document.getElementById('game');
const ctx = canvas.getContext('2d');
const scoreEl = document.getElementById('score');
const cellSize = 20;
const rows = canvas.height / cellSize;
const cols = canvas.width / cellSize;

let snake = [{x:5,y:5}];
let dir = {x:1,y:0};
let food = {};
let gameOver = false;
let score = 0;
let intervalId;

function placeFood() {
  let ok=false;
  while(!ok){
    const fx=Math.floor(Math.random()*cols);
    const fy=Math.floor(Math.random()*rows);
    if (!snake.some(s=>s.x===fx&&s.y===fy)) {food={x:fx,y:fy};ok=true;}
  }
}
function drawCell(x,y,color){ctx.fillStyle=color;ctx.fillRect(x*cellSize,y*cellSize,cellSize-1,cellSize-1);}
function update(){
  const head = {...snake[0]};
  head.x += dir.x;
  head.y += dir.y;

  // collision with walls
  if(head.x<0||head.x>=cols||head.y<0||head.y>=rows){end();return;}
  // collision with self
  if(snake.some(s=>s.x===head.x&&s.y===head.y)){end();return;}

  snake.unshift(head);
  if(head.x===food.x && head.y===food.y){
    score++;scoreEl.textContent=score;
    placeFood();
  }else{
    snake.pop();
  }
}
function draw(){
  ctx.clearRect(0,0,canvas.width,canvas.height);
  food && drawCell(food.x,food.y,'red');
  snake.forEach((s,i)=>drawCell(s.x,s.y,i===0?'lime':'green'));
}
function loop(){update();draw();}
function start(){
  snake=[{x:5,y:5}];
  dir={x:1,y:0};
  score=0;scoreEl.textContent=0;
  gameOver=false;
  placeFood();
  clearInterval(intervalId);
  intervalId=setInterval(loop,150);
}
function end(){gameOver=true;clearInterval(intervalId);alert('Game Over! Счёт '+score+'.\nНажмите любую клавишу для перезапуска.');}
document.addEventListener('keydown',e=>{
  if(gameOver){start();return;}
  const key=e.key;
  if(key==='ArrowUp'||key==='w'){if(dir.y!==1)dir={x:0,y:-1};}
  else if(key==='ArrowDown'||key==='s'){if(dir.y!==-1)dir={x:0,y:1};}
  else if(key==='ArrowLeft'||key==='a'){if(dir.x!==1)dir={x:-1,y:0};}
  else if(key==='ArrowRight'||key==='d'){if(dir.x!==-1)dir={x:1,y:0};}
});
start();
})();
</script>
</body>
</html>
```
