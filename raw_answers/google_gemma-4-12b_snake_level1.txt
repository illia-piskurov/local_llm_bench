```html
<!DOCTYPE html>
<html lang="ru">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Snake Game</title>
    <style>
        body {
            display: flex;
            justify-content: center;
            align-items: center;
            height: 100vh;
            margin: 0;
            background-color: #2c3e50;
            color: white;
            font-family: 'Arial', sans-serif;
            flex-direction: column;
        }

        #game-container {
            position: relative;
            border: 5px solid #ecf0f1;
            border-radius: 10px;
            box-shadow: 0 0 20px rgba(0,0,0,0.5);
        }

        canvas {
            display: block;
            background-color: #34495e;
        }

        #score-board {
            font-size: 24px;
            margin-bottom: 10px;
        }

        .game-over-overlay {
            position: absolute;
            top: 0;
            left: 0;
            width: 100%;
            height: 100%;
            background: rgba(0, 0, 0, 0.7);
            display: none;
            flex-direction: column;
            justify-content: center;
            align-items: center;
            border-radius: 5px;
        }

        h1 { margin: 0; font-size: 48px; color: #e74c3c; }
        p { font-size: 20px; margin: 10px 0; }
    </style>
</head>
<body>

    <div id="score-board">Счёт: <span id="score">0</span></div>

    <div id="game-container">
        <canvas id="snakeGame" width="400" height="400"></canvas>
        <div id="overlay" class="game-over-overlay">
            <h1>GAME OVER</h1>
            <p>Нажмите любую клавишу для рестарта</p>
        </div>
    </div>

    <script>
        const canvas = document.getElementById('snakeGame');
        const ctx = canvas.getContext('2d');
        const scoreElement = document.getElementById('score');
        const overlay = document.getElementById('overlay');

        const box = 20; // Размер одной ячейки
        let score = 0;
        let snake = [];
        let food = {};
        let d = null;
        let nextD = null;
        let gameActive = true;
        let gameLoop;

        // Инициализация игры
        function init() {
            snake = [
                {x: 9 * box, y: 10 * box},
                {x: 8 * box, y: 10 * box},
                {x: 7 * box, y: 10 * box}
            ];
            score = 0;
            scoreElement.innerText = score;
            d = "RIGHT";
            nextD = "RIGHT";
            gameActive = true;
            overlay.style.display = 'none';
            createFood();
        }

        function createFood() {
            food = {
                x: Math.floor(Math.random() * 19 + 1) * box,
                y: Math.floor(Math.random() * 19 + 1) * box
            };
            // Проверка, чтобы еда не появилась внутри змейки
            for (let i = 0; i < snake.length; i++) {
                if (snake[i].x === food.x && snake[i].y === food.y) {
                    createFood();
                    break;
                }
            }
        }

        document.addEventListener('keydown', direction);

        function direction(event) {
            const key = event.key;
            // Управление стрелками и WASD
            if ((key === 'ArrowUp' || key === 'w') && d !== "DOWN") nextD = "UP";
            else if ((key === 'ArrowDown' || key === 's') && d !== "UP") nextD = "DOWN";
            else if ((key === 'ArrowLeft' || key === 'a') && d !== "RIGHT") nextD = "LEFT";
            else if ((key === 'ArrowRight' || key === 'd') && d !== "LEFT") nextD = "RIGHT";

            // Рестарт при Game Over
            if (!gameActive && (key.length === 1)) {
                init();
            }
        }

        function draw() {
            if (!gameActive) return;

            d = nextD;
            ctx.fillStyle = "#34495e";
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // Отрисовка змейки
            for (let i = 0; i < snake.length; i++) {
                ctx.fillStyle = (i === 0) ? "#2ecc71" : "#27ae60";
                ctx.fillRect(snake[i].x, snake[i].y, box - 2, box - 2);
            }

            // Отрисовка еды
            ctx.fillStyle = "#e74c3c";
            ctx.fillRect(food.x, food.y, box - 2, box - 2);

            let snakeX = snake[0].x;
            let snakeY = snake[0].y;

            if (d === "LEFT") snakeX -= box;
            if (d === "UP") snakeY -= box;
            if (d === "RIGHT") snakeX += box;
            if (d === "DOWN") snakeY += box;

            // Проверка на поедание еды
            let ate = false;
            if (snakeX === food.x && snakeY === food.y) {
                score++;
                scoreElement.innerText = score;
                ate = true;
                createFood();
            }

            const newHead = { x: snakeX, y: snakeY };

            // Проверка столкновений (стены или хвост)
            if (
                snakeX < 0 || snakeX >= canvas.width ||
                snakeY < 0 || snakeY >= canvas.height ||
                snake.some((seg, index) => index !== snake.length - 1 && seg.x === snakeX && seg.y === snakeY)
            ) {
                gameActive = false;
                overlay.style.display = 'flex';
            }

            if (!ate) {
                snake.pop();
            }

            snake.unshift(newHead);
        }

        // Запуск цикла игры (100мс на шаг - скорость змейки)
        init();
        setInterval(draw, 100);
    </script>
</body>
</html>
```
